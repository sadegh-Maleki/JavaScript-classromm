array destructuring

    // before  destructuring
    const arr = [1, 2, 3, 4];
    const first = arr[0];
    const second = arr[1];
    const third = arr[2];
    const fourth = arr[3];

    // after  destructuring
const arr = [1, 2, 3, 4];
const [first,second,third,fourth] = arr;

console.log(first); // Outputs 1
console.log(second); // Outputs 2
console.log(third); // Outputs 3
console.log(fourth); // Outputs 4
